cmake_minimum_required(VERSION 3.12)
project(Sofa.GL.Component.Model LANGUAGES CXX)

set(SOFAGLCOMPONENTMODEL_SOURCE_DIR "src/sofa/gl/component/model")

set(HEADER_FILES
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/config.h.in
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/init.h
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/DataDisplay.h
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/MergeVisualModels.h
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/OglCylinderModel.h
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/OglGrid.h
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/OglLineAxis.h
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/OglModel.h
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/OglSceneFrame.h
	${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/PointSplatModel.h
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/SlicedVolumetricModel.h
)

set(SOURCE_FILES
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/init.cpp
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/DataDisplay.cpp
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/MergeVisualModels.cpp
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/OglCylinderModel.cpp
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/OglGrid.cpp
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/OglLineAxis.cpp
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/OglModel.cpp
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/OglSceneFrame.cpp
	${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/PointSplatModel.cpp
    ${SOFAGLCOMPONENTMODEL_SOURCE_DIR}/SlicedVolumetricModel.cpp
)

sofa_find_package(Sofa.GL REQUIRED)
sofa_find_package(Sofa.Component.Visual REQUIRED) # for VisualModelImpl
sofa_find_package(Sofa.GL.Component.OnScreen REQUIRED) # for OglColoMap in DataDisplay

add_library(${PROJECT_NAME} SHARED ${HEADER_FILES} ${SOURCE_FILES})
target_link_libraries(${PROJECT_NAME} PUBLIC Sofa.Component.Visual Sofa.GL.Component.OnScreen Sofa.GL)

sofa_create_package_with_targets(
    PACKAGE_NAME ${PROJECT_NAME}
    PACKAGE_VERSION ${Sofa_VERSION}
    TARGETS ${PROJECT_NAME} AUTO_SET_TARGET_PROPERTIES
    INCLUDE_SOURCE_DIR "src"
    INCLUDE_INSTALL_DIR "${PROJECT_NAME}"
)
