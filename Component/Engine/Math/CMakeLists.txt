cmake_minimum_required(VERSION 3.12)
project(Sofa.Component.Engine.Math LANGUAGES CXX)

set(SOFACOMPONENTENGINEMATH_SOURCE_DIR "src/sofa/component/engine/math")

set(HEADER_FILES
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/config.h.in
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/init.h
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/AverageCoord.h
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/AverageCoord.inl
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/DifferenceEngine.h
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/DifferenceEngine.inl
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/MathOp.h
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/MathOp.inl
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/SumEngine.h
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/SumEngine.inl
)

set(SOURCE_FILES
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/init.cpp
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/AverageCoord.cpp
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/DifferenceEngine.cpp
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/MathOp.cpp
    ${SOFACOMPONENTENGINEMATH_SOURCE_DIR}/SumEngine.cpp
)

sofa_find_package(SofaSimulationCore REQUIRED)

add_library(${PROJECT_NAME} SHARED ${HEADER_FILES} ${SOURCE_FILES})
target_link_libraries(${PROJECT_NAME} PUBLIC SofaSimulationCore)

sofa_create_package_with_targets(
    PACKAGE_NAME ${PROJECT_NAME}
    PACKAGE_VERSION ${Sofa_VERSION}
    TARGETS ${PROJECT_NAME} AUTO_SET_TARGET_PROPERTIES
    INCLUDE_SOURCE_DIR "src"
    INCLUDE_INSTALL_DIR "${PROJECT_NAME}"
)
