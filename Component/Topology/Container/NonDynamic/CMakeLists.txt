cmake_minimum_required(VERSION 3.12)
project(Sofa.Component.Topology.Container.NonDynamic LANGUAGES CXX)

set(SOFACOMPONENTTOPOLOGYCONTAINERNONDYNAMIC_SOURCE_DIR "src/sofa/component/topology/container/nondynamic")

set(HEADER_FILES
    ${SOFACOMPONENTTOPOLOGYCONTAINERNONDYNAMIC_SOURCE_DIR}/config.h.in
    ${SOFACOMPONENTTOPOLOGYCONTAINERNONDYNAMIC_SOURCE_DIR}/init.h
    ${SOFACOMPONENTTOPOLOGYCONTAINERNONDYNAMIC_SOURCE_DIR}/MeshTopology.h
    ${SOFACOMPONENTTOPOLOGYCONTAINERNONDYNAMIC_SOURCE_DIR}/CubeTopology.h
    ${SOFACOMPONENTTOPOLOGYCONTAINERNONDYNAMIC_SOURCE_DIR}/SphereQuadTopology.h
)

set(SOURCE_FILES
    ${SOFACOMPONENTTOPOLOGYCONTAINERNONDYNAMIC_SOURCE_DIR}/init.cpp
    ${SOFACOMPONENTTOPOLOGYCONTAINERNONDYNAMIC_SOURCE_DIR}/MeshTopology.cpp
    ${SOFACOMPONENTTOPOLOGYCONTAINERNONDYNAMIC_SOURCE_DIR}/CubeTopology.cpp
    ${SOFACOMPONENTTOPOLOGYCONTAINERNONDYNAMIC_SOURCE_DIR}/SphereQuadTopology.cpp
)

sofa_find_package(SofaSimulationCore REQUIRED)

add_library(${PROJECT_NAME} SHARED ${HEADER_FILES} ${SOURCE_FILES})
target_link_libraries(${PROJECT_NAME} PUBLIC SofaSimulationCore)

sofa_create_package_with_targets(
    PACKAGE_NAME ${PROJECT_NAME}
    PACKAGE_VERSION ${Sofa_VERSION}
    TARGETS ${PROJECT_NAME} AUTO_SET_TARGET_PROPERTIES
    INCLUDE_SOURCE_DIR "src"
    INCLUDE_INSTALL_DIR "${PROJECT_NAME}"
)
