cmake_minimum_required(VERSION 3.12)
project(Sofa.Component.UserInteraction.Controller LANGUAGES CXX)

set(SOFACOMPONENTUSERINTERACTIONCONTROLLER_SOURCE_DIR "src/sofa/component/userinteraction/controller")

set(HEADER_FILES
    ${SOFACOMPONENTUSERINTERACTIONCONTROLLER_SOURCE_DIR}/config.h.in
    ${SOFACOMPONENTUSERINTERACTIONCONTROLLER_SOURCE_DIR}/init.h    
    ${SOFACOMPONENTUSERINTERACTIONCONTROLLER_SOURCE_DIR}/MechanicalStateController.h
    ${SOFACOMPONENTUSERINTERACTIONCONTROLLER_SOURCE_DIR}/MechanicalStateController.inl
    ${SOFACOMPONENTUSERINTERACTIONCONTROLLER_SOURCE_DIR}/Controller.h
)

set(SOURCE_FILES
    ${SOFACOMPONENTUSERINTERACTIONCONTROLLER_SOURCE_DIR}/init.cpp
    ${SOFACOMPONENTUSERINTERACTIONCONTROLLER_SOURCE_DIR}/Controller.cpp
    ${SOFACOMPONENTUSERINTERACTIONCONTROLLER_SOURCE_DIR}/MechanicalStateController.cpp
)

sofa_find_package(Sofa.SimulationCore REQUIRED)

add_library(${PROJECT_NAME} SHARED ${HEADER_FILES} ${SOURCE_FILES})
target_link_libraries(${PROJECT_NAME} PUBLIC Sofa.SimulationCore)

sofa_create_package_with_targets(
    PACKAGE_NAME ${PROJECT_NAME}
    PACKAGE_VERSION ${Sofa_VERSION}
    TARGETS ${PROJECT_NAME} AUTO_SET_TARGET_PROPERTIES
    INCLUDE_SOURCE_DIR "src"
    INCLUDE_INSTALL_DIR "${PROJECT_NAME}"
)
