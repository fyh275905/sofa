cmake_minimum_required(VERSION 3.12)
project(Sofa.Component.UserInteraction.ConfigurationSetting LANGUAGES CXX)

set(SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR "src/sofa/component/userinteraction/configurationsetting")

set(HEADER_FILES
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/config.h.in
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/init.h
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/AddFrameButtonSetting.h
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/AddRecordedCameraButtonSetting.h
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/AttachBodyButtonSetting.h
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/BackgroundSetting.h
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/FixPickedParticleButtonSetting.h
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/MouseButtonSetting.h
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/SofaDefaultPathSetting.h
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/StatsSetting.h
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/ViewerSetting.h
)

set(SOURCE_FILES
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/init.cpp    
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/AddFrameButtonSetting.cpp
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/AddRecordedCameraButtonSetting.cpp
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/AttachBodyButtonSetting.cpp
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/BackgroundSetting.cpp
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/FixPickedParticleButtonSetting.cpp
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/MouseButtonSetting.cpp
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/SofaDefaultPathSetting.cpp
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/StatsSetting.cpp
    ${SOFACOMPONENTUSERINTERACTIONCONFIGURATIONSETTING_SOURCE_DIR}/ViewerSetting.cpp
)

sofa_find_package(Sofa.SimulationCore REQUIRED)

add_library(${PROJECT_NAME} SHARED ${HEADER_FILES} ${SOURCE_FILES})
target_link_libraries(${PROJECT_NAME} PUBLIC Sofa.SimulationCore)

sofa_create_package_with_targets(
    PACKAGE_NAME ${PROJECT_NAME}
    PACKAGE_VERSION ${Sofa_VERSION}
    TARGETS ${PROJECT_NAME} AUTO_SET_TARGET_PROPERTIES
    INCLUDE_SOURCE_DIR "src"
    INCLUDE_INSTALL_DIR "${PROJECT_NAME}"
)
