cmake_minimum_required(VERSION 3.12)
project(Sofa.Component.Utility LANGUAGES CXX)

set(SOFACOMPONENTUTILITY_SOURCE_DIR "src/sofa/component/utility")

set(HEADER_FILES
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/config.h.in
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/init.h
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/AddResourceRepository.h
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/APIVersion.h
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/Gravity.h
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/InfoComponent.h
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/MakeAliasComponent.h
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/MakeDataAliasComponent.h
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/MessageHandlerComponent.h
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/PauseAnimation.h
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/PauseAnimationOnEvent.h
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/RequiredPlugin.h
)

set(SOURCE_FILES
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/init.cpp
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/AddResourceRepository.cpp
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/APIVersion.cpp
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/Gravity.cpp
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/InfoComponent.cpp
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/MakeAliasComponent.cpp
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/MakeDataAliasComponent.cpp
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/MessageHandlerComponent.cpp
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/PauseAnimation.cpp
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/PauseAnimationOnEvent.cpp
    ${SOFACOMPONENTUTILITY_SOURCE_DIR}/RequiredPlugin.cpp
)

sofa_find_package(SofaSimulationCore REQUIRED)

add_library(${PROJECT_NAME} SHARED ${HEADER_FILES} ${SOURCE_FILES})
target_link_libraries(${PROJECT_NAME} PUBLIC SofaSimulationCore)

sofa_create_package_with_targets(
    PACKAGE_NAME ${PROJECT_NAME}
    PACKAGE_VERSION ${Sofa_VERSION}
    TARGETS ${PROJECT_NAME} AUTO_SET_TARGET_PROPERTIES
    INCLUDE_SOURCE_DIR "src"
    INCLUDE_INSTALL_DIR "${PROJECT_NAME}"
)
